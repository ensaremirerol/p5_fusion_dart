String get name {
  switch (this) {
    case Stat.strength:
      return "Strength";
    case Stat.magic:
      return "Magic";
    case Stat.endurance:
      return "Endurance";
    case Stat.agility:
      return "Agility";
    case Stat.luck:
      return "Luck";
  }
}
